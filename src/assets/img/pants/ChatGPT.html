To implement pagination with 8 items per page, you can create a parent component that manages the pagination and passes the appropriate items to your Productitems component. Here's an example of how you can do this in React:

  1. Create a parent component for pagination. This component will manage the items to be displayed on each page.
jsx
import React, { useState } from 'react';
import Productitems from './Productitems';

const Pagination = ({ items }) => {
  const itemsPerPage = 8;
  const [currentPage, setCurrentPage] = useState(1);

  const indexOfLastItem = currentPage * itemsPerPage;
  const indexOfFirstItem = indexOfLastItem - itemsPerPage;
  const currentItems = items.slice(indexOfFirstItem, indexOfLastItem);

  const paginate = (pageNumber) => {
    setCurrentPage(pageNumber);
  };

  return (
    <div>
      {currentItems.map((item) => (
        <Productitems key={item.id} item={item} />
      )}

      <ul className="pagination">
        {Array.from({ length: Math.ceil(items.length / itemsPerPage) }).map(
          (_, index) => (
            <li key={index}>
              <a
                href="#"
                onClick={() => paginate(index + 1)}
              >
                {index + 1}
              </a>
            </li>
          )
        )}
      </ul>
    </div>
  );
};

export default Pagination;
  1. Update your Productitems component to receive the item and render it as before:
jsx
import React from 'react';
import ShoppingCartIcon from '@mui/icons-material/ShoppingCart';
import Modal from '../components/Modal';
import { useDispatch, useSelector } from 'react-redux';
import { addToCart, removeFromCart } from '../redux/CartSlice';

const Productitems = ({ item }) => {
  const cart = useSelector((state) => state.cart.cart);
  const selectedCategory = useSelector((state) => state.cart.selectedCategory);
  const selectedPrice = useSelector((state) => state.cart.selectedPrice);
  const dispatch = useDispatch();

  const addItemToCart = (item) => {
    dispatch(addToCart(item));
  };

  const removeItemFromCart = (item) => {
    dispatch(removeFromCart(item));
  };

  if (
    (selectedCategory && item.category !== selectedCategory) ||
    (selectedPrice && item.price !== selectedPrice)
  ) {
    return null;
  }

  return (
    <div className="prods">
      <div className="product-item">
        <div>
          <div className="product-img">
            <img src={item.image} alt={item.title} className="product--img" />
          </div>
          <div>
            <div className="modal">
              <p className="product-price">
                <strong>Price:</strong> ${item.price}
              </p>
              <Modal item={item} />
            </div>
          </div>
          <div className="product-details">
            {cart.some((x) => x.id === item.id) ? (
              <p onClick={() => removeItemFromCart(item)} className="product-btn-cart">
                Remove from cart
                <ShoppingCartIcon style={{ width: 15 }} />
              </p>
            ) : (
              <p onClick={() => addItemToCart(item)} className="product-btn-cart">
                Add to cart
                <ShoppingCartIcon style={{ width: 15 }} />
              </p>
            )}
          </div>
        </div>
      </div>
    </div>
  );
};

export default Productitems;
